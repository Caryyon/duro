import { Meta, Story, Canvas } from '@storybook/addon-docs'
import { Badge, Separator, Avatar, AvatarImage, AvatarFallback, Skeleton, Card, CardContent, Stack, Cluster, Box } from '@caryyon/duro'

export const BadgeTemplate = (args) => <Badge {...args} />

export const AvatarTemplate = (args) => (
  <Avatar {...args}>
    {args.showImage && <AvatarImage src={args.imageUrl} alt="User" />}
    <AvatarFallback>{args.fallbackText}</AvatarFallback>
  </Avatar>
)

export const SkeletonTemplate = (args) => <Skeleton {...args} />
