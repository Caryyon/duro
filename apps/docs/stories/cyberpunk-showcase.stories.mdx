export const GaugeMeter = ({ value = 65, size = 100 }) => {
  const angle = (value / 100) * 180 - 90
  const radius = size / 2 - 10
  return (
    <div style={{ width: size, height: size / 2 + 20, position: 'relative' }}>
      <svg width={size} height={size / 2 + 20}>
        {/* Background arc */}
        <path
          d={`M10,${size/2} A${radius},${radius} 0 0,1 ${size - 10},${size/2}`}
          fill="none"
          stroke="var(--duro-color-secondary)"
          strokeWidth="4"
          strokeDasharray="4 2"
          opacity="0.5"
        />
        {/* Value arc */}
        <path
          d={`M10,${size/2} A${radius},${radius} 0 0,1 ${10 + radius * (1 - Math.cos(angle * Math.PI / 180))},${size/2 - radius * Math.sin(angle * Math.PI / 180)}`}
          fill="none"
          stroke="var(--duro-color-text)"
          strokeWidth="4"
        />
        {/* Tick marks */}
        {[0, 25, 50, 75, 100].map((tick, i) => {
          const tickAngle = (tick / 100) * 180 - 90
          const x1 = size/2 + (radius - 8) * Math.cos(tickAngle * Math.PI / 180)
          const y1 = size/2 - (radius - 8) * Math.sin(tickAngle * Math.PI / 180)
          const x2 = size/2 + (radius + 2) * Math.cos(tickAngle * Math.PI / 180)
          const y2 = size/2 - (radius + 2) * Math.sin(tickAngle * Math.PI / 180)
          return <line key={i} x1={x1} y1={y1} x2={x2} y2={y2} stroke="var(--duro-color-text)" strokeWidth="1" />
        })}
        {/* Center point */}
        <circle cx={size/2} cy={size/2} r="4" fill="var(--duro-color-text)" />
        {/* Needle */}
        <line
          x1={size/2}
          y1={size/2}
          x2={size/2 + (radius - 15) * Math.cos(angle * Math.PI / 180)}
          y2={size/2 - (radius - 15) * Math.sin(angle * Math.PI / 180)}
          stroke="var(--duro-color-text)"
          strokeWidth="2"
        />
      </svg>
      <div style={{
        position: 'absolute',
        bottom: 0,
        left: '50%',
        transform: 'translateX(-50%)',
        fontFamily: 'var(--duro-font-monospace)',
        fontSize: 'var(--duro-font-size-lg)',
        fontWeight: 300
      }}>{value}%</div>
    </div>
  )
}
