## Complete Form Example

This example showcases a polished registration form with proper visual hierarchy,
breathing room between elements, and correctly aligned action buttons.

<Canvas>
  <Story name="Complete Form">
    <Box maxWidth="sm" centered padding="lg">
      <Card>
        <CardHeader>
          <CardTitle>Create Account</CardTitle>
        </CardHeader>
        <CardContent>
          <Stack as="form" variant="loose">
            <Grid variant="cols2">
              <Stack>
                <Label required>First Name</Label>
                <Input placeholder="John" fullWidth />
              </Stack>
              <Stack>
                <Label required>Last Name</Label>
                <Input placeholder="Doe" fullWidth />
              </Stack>
            </Grid>
            <Stack>
              <Label required>Email</Label>
              <Input type="email" placeholder="john@example.com" fullWidth />
            </Stack>
            <Stack>
              <Label required>Password</Label>
              <Input type="password" placeholder="••••••••" fullWidth />
            </Stack>
            <Stack>
              <Label>Bio</Label>
              <Textarea placeholder="Tell us about yourself..." rows={3} fullWidth />
            </Stack>
            <Stack variant="tight">
              <Cluster>
                <Checkbox />
                <span>I agree to the terms and conditions</span>
              </Cluster>
              <Cluster>
                <Switch />
                <span>Send me email notifications</span>
              </Cluster>
            </Stack>
          </Stack>
        </CardContent>
        <CardFooter justify="spaceBetween" bordered>
          <Button variant="ghost">Cancel</Button>
          <Button variant="primary">Create Account</Button>
        </CardFooter>
      </Card>
    </Box>
  </Story>
</Canvas>
