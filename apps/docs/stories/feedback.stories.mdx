import { Meta, Story, Canvas } from '@storybook/addon-docs'
import { Alert, AlertTitle, AlertDescription, Progress, Tooltip, Button, Stack, Cluster, Box } from '@caryyon/duro'

export const AlertTemplate = (args) => (
  <Alert variant={args.variant}>
    {args.showTitle && <AlertTitle>{args.title}</AlertTitle>}
    <AlertDescription>{args.description}</AlertDescription>
  </Alert>
)

export const ProgressTemplate = (args) => <Progress {...args} />

export const TooltipTemplate = (args) => (
  <Tooltip content={args.content} position={args.position} delay={args.delay}>
    <Button>{args.buttonText}</Button>
  </Tooltip>
)
