import { Meta, Story, Canvas } from '@storybook/addon-docs'
import { Card, CardHeader, CardTitle, CardDescription, CardContent, CardFooter, Button, Stack, Grid, Cluster, Checkbox } from '@caryyon/duro'

export const CardTemplate = (args) => (
  <Card>
    {args.showHeader && (
      <CardHeader>
        <CardTitle>{args.title}</CardTitle>
        {args.showDescription && <CardDescription>{args.description}</CardDescription>}
      </CardHeader>
    )}
    <CardContent>
      <p>{args.content}</p>
    </CardContent>
    {args.showFooter && (
      <CardFooter>
        <Button variant="primary">Action</Button>
        <Button variant="secondary">Cancel</Button>
      </CardFooter>
    )}
  </Card>
)

<Meta title="Components/Card" component={Card} />

# Card

A versatile container component with header, content, and footer sections.
