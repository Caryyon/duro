export const ButtonGroupTemplate = (args) => (
  <ButtonGroup orientation={args.orientation} attached={args.attached}>
    <Button>{args.button1Text}</Button>
    <Button>{args.button2Text}</Button>
    <Button>{args.button3Text}</Button>
  </ButtonGroup>
)

export const AccordionTemplate = (args) => (
  <Accordion type={args.type}>
    <AccordionItem value="item1">
      <AccordionTrigger>{args.item1Title}</AccordionTrigger>
      <AccordionContent>{args.item1Content}</AccordionContent>
    </AccordionItem>
    <AccordionItem value="item2">
      <AccordionTrigger>{args.item2Title}</AccordionTrigger>
      <AccordionContent>{args.item2Content}</AccordionContent>
    </AccordionItem>
    <AccordionItem value="item3">
      <AccordionTrigger>{args.item3Title}</AccordionTrigger>
      <AccordionContent>{args.item3Content}</AccordionContent>
    </AccordionItem>
  </Accordion>
)
